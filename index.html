-- Corre isto no SQL Editor do Supabase para criar as facturas de clientes. create table if not exists public.factures_vente ( id uuid primary key default gen_random_uuid(), numero text not null unique, client_id uuid references public.clients(id), chantier_id uuid references public.chantiers_lyneo(id), devis_id uuid references public.devis(id), type_facture text not null default 'situation', -- acompte, situation, solde, libre date_creation date default current_date, date_echeance date, total_ht numeric default 0, total_tva numeric default 0, total_ttc numeric default 0, statut text not null default 'brouillon', -- brouillon, envoyee, payee_partiel, payee, annulee observations text, cree_par uuid references public.utilisateurs(id), created_at timestamptz default now() ); create table if not exists public.lignes_facture_vente ( id uuid primary key default gen_random_uuid(), facture_vente_id uuid not null references public.factures_vente(id) on delete cascade, ligne_devis_id uuid references public.lignes_devis(id), -- null si ligne ajoutée manuellement (hors devis) ordre int not null default 1, description text not null, unite text default 'u', quantite numeric default 1, prix_unitaire_ht numeric default 0, taux_tva numeric default 17, created_at timestamptz default now() ); create index if not exists idx_factures_vente_chantier_id on public.factures_vente(chantier_id); create index if not exists idx_factures_vente_client_id on public.factures_vente(client_id); create index if not exists idx_factures_vente_devis_id on public.factures_vente(devis_id); create index if not exists idx_lignes_facture_vente_facture_id on public.lignes_facture_vente(facture_vente_id); create index if not exists idx_lignes_facture_vente_ligne_devis_id on public.lignes_facture_vente(ligne_devis_id); create or replace function generate_facture_vente_number() returns text language plpgsql as $$ declare year_part text := to_char(current_date, 'YYYY'); seq_num int; begin select coalesce(max(cast(split_part(numero,'-',3) as int)),0)+1 into seq_num from public.factures_vente where numero like 'FC-'||year_part||'-%'; return 'FC-'||year_part||'-'||lpad(seq_num::text,3,'0'); end; $$; alter table public.factures_vente enable row level security; alter table public.lignes_facture_vente enable row level security; drop policy if exists "public_full_access_factures_vente" on public.factures_vente; create policy "public_full_access_factures_vente" on public.factures_vente for all to public using (true) with check (true); drop policy if exists "public_full_access_lignes_facture_vente" on public.lignes_facture_vente; create policy "public_full_access_lignes_facture_vente" on public.lignes_facture_vente for all to public using (true) with check (true);
