-- Corre isto no SQL Editor do Supabase para criar o módulo Devis. create table if not exists public.devis ( id uuid primary key default gen_random_uuid(), numero text not null unique, client_id uuid references public.clients(id), chantier_id uuid references public.chantiers_lyneo(id), objet text, correspondant text, date_devis date default current_date, validite_jours int default 30, statut text not null default 'brouillon', -- brouillon, envoye, accepte, refuse, expire conditions_acompte_pct numeric default 40, conditions_paiement text default '30 jours', montant_ht numeric default 0, montant_tva numeric default 0, montant_ttc numeric default 0, observations text, cree_par uuid references public.utilisateurs(id), created_at timestamptz default now() ); create table if not exists public.lignes_devis ( id uuid primary key default gen_random_uuid(), devis_id uuid not null references public.devis(id) on delete cascade, bloc text not null default 'principal', -- 'principal' ou 'option' type text not null default 'ligne', -- 'section' ou 'ligne' ordre int not null default 1, description text not null, unite text default 'u', quantite numeric default 1, prix_unitaire_ht numeric default 0, taux_tva numeric default 17, created_at timestamptz default now() ); create table if not exists public.anexos_devis ( id uuid primary key default gen_random_uuid(), devis_id uuid not null references public.devis(id) on delete cascade, nome_ficheiro text, url text, tamanho bigint, tipo text, carregado_por uuid references public.utilisateurs(id), created_at timestamptz default now() ); create index if not exists idx_lignes_devis_devis_id on public.lignes_devis(devis_id); create index if not exists idx_anexos_devis_devis_id on public.anexos_devis(devis_id); create index if not exists idx_devis_client_id on public.devis(client_id); create index if not exists idx_devis_chantier_id on public.devis(chantier_id); -- Função para gerar o número no formato DV-2026-001 (igual ao teu modelo atual) create or replace function generate_devis_number() returns text language plpgsql as $$ declare year_part text := to_char(current_date, 'YYYY'); seq_num int; begin select coalesce(max(cast(split_part(numero,'-',3) as int)),0)+1 into seq_num from public.devis where numero like 'DV-'||year_part||'-%'; return 'DV-'||year_part||'-'||lpad(seq_num::text,3,'0'); end; $$; alter table public.devis enable row level security; alter table public.lignes_devis enable row level security; alter table public.anexos_devis enable row level security; -- Política "to public" porque a app não faz login oficial do Supabase Auth -- (usa sempre o papel anon/public, como já identificámos) drop policy if exists "public_full_access_devis" on public.devis; create policy "public_full_access_devis" on public.devis for all to public using (true) with check (true); drop policy if exists "public_full_access_lignes_devis" on public.lignes_devis; create policy "public_full_access_lignes_devis" on public.lignes_devis for all to public using (true) with check (true); drop policy if exists "public_full_access_anexos_devis" on public.anexos_devis; create policy "public_full_access_anexos_devis" on public.anexos_devis for all to public using (true) with check (true);
